ENTRADA:

- items: lista de numeros enteros a modificar

SALIDA:

- items: lista modificada

N = len(items)

for i in range(0, N - 1):
    for j in range(0, N - 1 - i):
        if items[j] > items[j + 1]:
            temp = items[j]
            items[j] = items[j + 1]
            items[j + 1] = temp

return items
